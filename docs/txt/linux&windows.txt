#windows快捷键
ctrl + insert 	复制选中内容
shift + end		选中从光标处到行尾内容
shift + home	选中从行首到该光标处的内容
Ctrl + Home		跳转到文件的首行
Ctrl + End 		跳转到文件的最后一行


#bash、vi下通用快捷键
shift + insert 	粘贴


#bash快捷键
	ctrl+a		移到行首
	ctrl+e		移到行尾
	ctrl+w		删除光标前一个单词
	ctrl+u		删除光标前所有字符
	ctrl+l		清屏
	ctrl+k		删除光标后所有字符	
	ctrl+p		显示上一个输入的命令
	ctrl+n		显示下一个输入的命令		
	ctrl+r		根据关键词查找历史命令
	cd -		快速切换两个目录

	
#vi快捷键
	yw			复制单词
	y$			复制到行尾
	p 			粘贴
	-------------------------
	:s/old/new			替换当前行中的数据
	:s/old/new/g 		替换当前行中的数据
	:n,ms/old/new/g		替换n和m行之间的所有old
	:%s/old/new/g		替换整个文件中的所有old
	:%s/old/new/gc		替换整个文件中的所有old，但在每次出现时都提示
	-------------------------
	v			进入可视模式
				选中要复制的字符
	y			复制
	p 			粘贴
	-------------------------
	u			撤销
	ctrl+r		重做
	x			删除当前光标下的行
	ctrl+b		向上移动一屏
	PageUp		向上移动一屏
	ctrl+f      向下移动一屏
	PageDown	向下移动一屏
	:1,$d		全部删除
	：35,69 de 	删除35到69行
	:n			定位到某行
	:w a.txt	保存到a.txt
	shift+4		移动到行尾
	shift+6		移动到行首
	shift+g		跳到最后一行
	G			跳到最后一行
	gg			跳到第一行
	yy			复制
	p			粘贴
	h			左移一个字符
	j			下移一个字符
	k			上移一个字符
	l			右移一个字符
	dd			删除当前行
	dw 			删除当前单词
	d$			删除光标位置到行尾内容
	J			删除光标到行行尾的换行符
	a			当前光标后追加数据
	A			在当前光标所在行行尾追加数据
